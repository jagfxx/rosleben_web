# Nginx — ROSLEBEN Landing
# Copiar en: /etc/nginx/sites-available/rosleben-web
# Dominio: rosleben.com

# HTTP → HTTPS
server {
    listen 80;
    server_name rosleben.com www.rosleben.com;

    return 301 https://$server_name$request_uri;
}

# HTTPS
server {
    listen 443 ssl http2;
    server_name rosleben.com www.rosleben.com;

    ssl_certificate /etc/letsencrypt/live/rosleben.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/rosleben.com/privkey.pem;
    include /etc/letsencrypt/options-ssl-nginx.conf;
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem;

    add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
    add_header X-Frame-Options "SAMEORIGIN" always;
    add_header X-Content-Type-Options "nosniff" always;

    access_log /var/log/nginx/rosleben-web-access.log;
    error_log /var/log/nginx/rosleben-web-error.log;

    client_max_body_size 10M;

    # Puerto 3002 — TSM: 3000, THEPIOLO: 3001
    # Un solo location: bloques separados para /_next/static sin
    # proxy_set_header Host provocan 400 en chunks JS/CSS de Next.js.
    location / {
        proxy_pass http://localhost:3002;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }
}
